DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT 
    job.job_id,
    job.name AS job_name,
    job.enabled,
    job.description,
    job.date_created,
    job.date_modified
FROM msdb.dbo.sysjobs job;

DECLARE @job_name NVARCHAR(MAX), @enabled NVARCHAR(MAX), @date_created NVARCHAR(MAX), @createTime NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @job_name, @enabled, @date_created;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @job_name + ',' + @enabled + ',' + @date_created;

    FETCH NEXT FROM @job_name, @enabled, @date_created;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
