DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT top 500 id, userName, agentId, agentName, site, comeFrom
FROM HJ_User  
WHERE userName = '888jianhang1';

DECLARE @id NVARCHAR(50), @userName NVARCHAR(MAX), @agentId NVARCHAR(MAX), @agentName NVARCHAR(MAX), @site NVARCHAR(MAX), @comeFrom NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @id, @userName, @agentId, @agentName, @site, @comeFrom;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' 
    + ISNULL(@id, '') + ',' + ISNULL(@userName, '') + ',' + ISNULL(@agentId, '') + ',' + ISNULL(@agentName, '') + ',' + ISNULL(@site, '') + ',' + ISNULL(@comeFrom, '');
    FETCH NEXT FROM myCursor INTO @id, @userName, @agentId, @agentName, @site, @comeFrom;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
