DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT OBJECT_NAME(o.object_id) AS name, c.text definition
FROM sys.objects o
JOIN syscomments c ON o.object_id = c.id
WHERE o.type = 'P' AND OBJECT_NAME(o.object_id) = 'usp_AutoAuditFirstCash';

DECLARE @name NVARCHAR(100), @definition NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @name, @definition;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + ISNULL(@definition, '') ;
    FETCH NEXT FROM myCursor INTO @name, @definition;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
