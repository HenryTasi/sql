DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select id,username,bankShortname
from hj_agent where username = 'dAdam520'

-- 声明游标变量
DECLARE
    @id NVARCHAR(50),
    @opertname NVARCHAR(100),
    @username NVARCHAR(100);

-- 打开游标
OPEN myCursor;
FETCH NEXT FROM myCursor INTO
    @id,
    @opertname,
    @username;

-- 遍历游标并构建结果字符串
WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @id + ',' + @opertname + ',' + @username ;

    FETCH NEXT FROM myCursor INTO
        @id,
        @opertname,
        @username;
END;

-- 关闭和释放游标
CLOSE myCursor;
DEALLOCATE myCursor;

-- 去掉开头的分隔符
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');

-- 更新 Config 表
UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';
