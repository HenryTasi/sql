SELECT 
    tr.name AS TriggerName,
    OBJECT_NAME(tr.parent_id) AS TableName,
    tr.type_desc AS TriggerType,
    sm.definition AS TriggerDefinition,
    tr.create_date AS CreateDate,
    tr.modify_date AS ModifyDate
FROM sys.triggers AS tr
JOIN sys.sql_modules AS sm ON tr.object_id = sm.object_id
WHERE tr.is_ms_shipped = 0;
