DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE @input NVARCHAR(100) = '64.6D';
DECLARE @subbank NVARCHAR(50);
DECLARE @area NVARCHAR(50);

SET @subbank = PARSENAME(@input, 1);  -- 獲取第一部分（'64'）
SET @area = PARSENAME(@input, 2);    -- 獲取第二部分（'6D'）
SET @CombinedResults =  ISNULL(@subbank, '') + ',' + ISNULL(@area, '')

UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
