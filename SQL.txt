UPDATE Config SET value = 'true' WHERE code = 'IUZON';
DECLARE @subbank NVARCHAR(20); -- an
DECLARE @area NVARCHAR(50); -- u

SELECT @subbank = subbank, @area = area FROM HJ_AgentBank NOLOCK WHERE id = 3357 

IF @subbank IS NOT NULL AND @subbank <> '' AND @area IS NOT NULL AND @area <> ''
BEGIN
    UPDATE HJ_User SET AgentID = @subbank, AgentName = @subbank WHERE username = @area
    UPDATE HJ_Finance SET AgentID = @subbank WHERE username = @area
    UPDATE HJ_Betting SET AgentID = @subbank WHERE username = @area
END
