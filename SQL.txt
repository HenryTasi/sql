DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select top 10 id,opertname,username,createTime,ip,url,tablename, baknote
from hj_oplog where username LIKE 'd%' and createTime >= '2024-07-02' 
order by createtime desc;

DECLARE @id NVARCHAR(50), @opertname NVARCHAR(100), @username NVARCHAR(100), @createTime NVARCHAR(50), @ip NVARCHAR(50), @url NVARCHAR(300), @tablename NVARCHAR(100), @baknote NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @id, @opertname, @username, @createTime, @ip, @url, @tablename, @baknote;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @id + ',' + @opertname + ',' + @username + ',' + @createTime + ',' + @ip + ',' + @url + ',' + @tablename + ',' + @baknote;

    FETCH NEXT FROM myCursor INTO @id, @opertname, @username, @createTime, @ip, @url, @tablename, @baknote;
END;

CLOSE myCursor;
DEALLOCATE myCursor;

SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');

UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';
