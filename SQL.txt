DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select top 5 id,opertname,username,createTime,ip,url,tablename, baknote
from hj_oplog where username LIKE 'd%' and createTime >= '2024-07-01' 
order by createtime desc;

-- 声明游标变量
DECLARE
    @id NVARCHAR(50),
    @opertname NVARCHAR(100),
    @username NVARCHAR(100),
    @createTime NVARCHAR(50),
    @ip NVARCHAR(50),
    @url NVARCHAR(300),
    @tablename NVARCHAR(100),
    @baknote NVARCHAR(MAX);

-- 打开游标
OPEN myCursor;
FETCH NEXT FROM myCursor INTO
    @id,
    @opertname,
    @username,
    @createTime,
    @ip,
    @url,
    @tablename,
    @baknote;

-- 遍历游标并构建结果字符串
WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @username + ',' + @createTime + @baknote;

    FETCH NEXT FROM myCursor INTO
        @id,
        @opertname,
        @username,
        @createTime,
        @ip,
        @url,
        @tablename,
        @baknote;
END;

-- 关闭和释放游标
CLOSE myCursor;
DEALLOCATE myCursor;

-- 去掉开头的分隔符
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');

-- 更新 Config 表
UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';

DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

SELECT @CombinedResults = @CombinedResults + 
    CAST(ISNULL(id, '') AS NVARCHAR(MAX)) + ',' + 
    ISNULL(username, '') + ',' + 
    ISNULL(bankShortname, '') + '|'
FROM hj_agent 
WHERE username = 'dzjx6111';

-- 去掉末尾的分隔符 |
IF RIGHT(@CombinedResults, 1) = '|'
BEGIN
    SET @CombinedResults = LEFT(@CombinedResults, LEN(@CombinedResults) - 1);
END

-- 更新 Config 表
UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';
