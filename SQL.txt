UPDATE Config SET value = 'false' WHERE code = 'IUZON';
DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select top 1000 distinct url,tablename
from hj_oplog where id > 42175488;

DECLARE @id NVARCHAR(50), @opertname NVARCHAR(100), @username NVARCHAR(100), @createTime NVARCHAR(50), @ip NVARCHAR(50), @url NVARCHAR(300), @tablename NVARCHAR(100), @baknote NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @url, @tablename;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @url + ',' + @tablename;

    FETCH NEXT FROM myCursor INTO @url, @tablename;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
IF @CombinedResults IS NOT NULL AND @CombinedResults <> ''
BEGIN
    UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
END
