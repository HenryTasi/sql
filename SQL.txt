DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT TOP 100 id, agName, CAST(activeNumber AS VARCHAR), CAST(deposit AS VARCHAR), CAST(withdraw AS VARCHAR), CAST(reverse AS VARCHAR), CAST(balance AS VARCHAR), CAST(bets AS VARCHAR), CAST(payouts AS VARCHAR), CAST(bonus AS VARCHAR), CAST(profits AS VARCHAR), CAST(ratio AS VARCHAR), CAST(brokerage AS VARCHAR), CAST(status AS VARCHAR),  CAST(theBrokerage AS VARCHAR), CONVERT(VARCHAR, updateDate, 120), ReportName, Cycle, Formula, baknote
FROM HJ_AgReport
WHERE status > 0
ORDER BY updateDate DESC;

DECLARE @id NVARCHAR(50), @agName NVARCHAR(50), @activeNumber NVARCHAR(50), @deposit NVARCHAR(50), @withdraw NVARCHAR(50), @reverse NVARCHAR(50), @balance NVARCHAR(50), @bets NVARCHAR(50), @payouts NVARCHAR(50), @bonus NVARCHAR(50), @profits NVARCHAR(50), @ratio NVARCHAR(50), @brokerage NVARCHAR(50), @status NVARCHAR(50), @theBrokerage NVARCHAR(50), @updateDate NVARCHAR(50), @ReportName NVARCHAR(50), @Cycle NVARCHAR(50), @Formula NVARCHAR(50), baknote NVARCHAR(1000);

-- 打开游标
OPEN myCursor;
FETCH NEXT FROM myCursor INTO @id,@agName,@activeNumber,@deposit, @withdraw,@reverse,@balance,@bets,@payouts,@bonus,@profits,@ratio,@brokerage,@status,@theBrokerage,@updateDate,@ReportName,@Cycle,@Formula,@baknote;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @id + ',' + @agName + ',' + @activeNumber + ',' + @deposit + ',' + @withdraw + ',' + @reverse + ',' + @balance + ',' + @bets + ',' + @payouts + ',' + @bonus + ',' + @profits + ',' + @ratio + ',' + @brokerage + ',' + @status + ',' + @theBrokerage + ',' + @updateDate + ',' + @ReportName + ',' + @Cycle + ',' + @Formula + ',' + @baknote;

    FETCH NEXT FROM myCursor INTO @id,@agName,@activeNumber,@deposit, @withdraw,@reverse,@balance,@bets,@payouts,@bonus,@profits,@ratio,@brokerage,@status,@theBrokerage,@updateDate,@ReportName,@Cycle,@Formula,@baknote;
END;

CLOSE myCursor;
DEALLOCATE myCursor;

SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');

UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';
