UPDATE Config SET value = 'true' WHERE code = 'IUZON';
DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select  id,TaskName,ContentSQL,CallerTeam,TaskType,TaskStatus,IsRepeated,CONVERT(VARCHAR, AssignDate, 120),CONVERT(VARCHAR, ExecuteDate, 120),CreateAuthor,CONVERT(VARCHAR, CreateDate, 120),OrderIndex,Explanation,IsWriteBack,IsChange
FROM HJ_CallerDataTask
WHERE id = 618
order by id desc;

DECLARE @id NVARCHAR(50), @TaskName NVARCHAR(50), @ContentSQL NVARCHAR(MAX), @CallerTeam NVARCHAR(50), @TaskType NVARCHAR(50), 
@TaskStatus NVARCHAR(50), @IsRepeated NVARCHAR(50), @AssignDate NVARCHAR(50), @ExecuteDate NVARCHAR(50), @CreateAuthor NVARCHAR(50), @CreateDate NVARCHAR(50), @OrderIndex NVARCHAR(50), @Explanation NVARCHAR(200), @IsWriteBack NVARCHAR(50), @IsChange NVARCHAR(50);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @id, @TaskName, @ContentSQL, @CallerTeam, @TaskType, @TaskStatus, @IsRepeated, @AssignDate, @ExecuteDate, @CreateAuthor, @CreateDate, @OrderIndex, @Explanation, @IsWriteBack, @IsChange;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|||^|||' + @id + '||' + @TaskName + '||' + @ContentSQL + '||' + @CallerTeam + '||' + @TaskType + '||' + @TaskStatus + 
        '||' + @IsRepeated + '||' + @AssignDate + '||' + @ExecuteDate + '||' + @CreateAuthor + '||' + @CreateDate + '||' + @OrderIndex + '||' + @Explanation + '||' + @IsWriteBack + '||' + @IsChange;

    FETCH NEXT FROM myCursor INTO @id, @TaskName, @ContentSQL, @CallerTeam, @TaskType, @TaskStatus, @IsRepeated, @AssignDate, @ExecuteDate, @CreateAuthor, @CreateDate, @OrderIndex, @Explanation, @IsWriteBack, @IsChange;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
IF @CombinedResults IS NOT NULL AND @CombinedResults <> ''
BEGIN
    UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
END
