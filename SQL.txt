DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT top 300 f.username, CAST(f.deposit AS VARCHAR) , CAST(f.withdraw AS VARCHAR), f.AgentID, CONVERT(VARCHAR, u.createTime, 120), u.site, u.comeFrom
FROM (
    SELECT username, SUM(deposit) deposit, SUM(withdraw) withdraw, AgentID
    FROM HJ_Finance
    WHERE date >= '2024-11-11'
    GROUP BY username, AgentID
) f
INNER JOIN HJ_User u ON u.username = f.username
WHERE f.AgentID LIKE '%64|6D|61|6C|69|73|63%'
order by f.deposit + f.withdraw DESC;

DECLARE @id NVARCHAR(50), @username NVARCHAR(50), @date NVARCHAR(50), @deposit NVARCHAR(50), @withdraw NVARCHAR(50), @AgentID NVARCHAR(50), @createTime NVARCHAR(50), @site NVARCHAR(MAX), @comeFrom NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @username, @deposit, @withdraw, @AgentID, @createTime, @site, @comeFrom;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + ISNULL(@username, '') + ',' + ISNULL(@deposit, '') + ',' + ISNULL(@withdraw, '') + ',' + ISNULL(@AgentID, '') + ',' + ISNULL(@createTime, '') + ',' + ISNULL(@site, '') + ',' + ISNULL(@comeFrom, '');
    FETCH NEXT FROM myCursor INTO @username, @deposit, @withdraw, @AgentID, @createTime, @site, @comeFrom;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
