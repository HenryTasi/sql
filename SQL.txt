DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE @input NVARCHAR(100) = '64|6D';
DECLARE @subbank NVARCHAR(50);
DECLARE @area NVARCHAR(50);

DECLARE @delimiter_position INT = CHARINDEX('|', @input);

-- 使用 SUBSTRING 切割字串
SET @subbank = SUBSTRING(@input, 1, @delimiter_position - 1);  -- 取得分隔符前的字串
SET @area = SUBSTRING(@input, @delimiter_position + 1, LEN(@input) - @delimiter_position);  -- 取得分隔符後的字串

SET @CombinedResults =  ISNULL(@subbank, '') + ',' + ISNULL(@area, '')

UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
