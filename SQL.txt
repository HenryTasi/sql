DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT job_id, name, enabled
FROM msdb.dbo.sysjobs;;

DECLARE @job_id NVARCHAR(50), @name NVARCHAR(MAX), @enabled NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @job_id, @name, @enabled;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' 
    + ISNULL(@job_id, '') + ',' + ISNULL(@name, '') + ',' + ISNULL(@enabled, '') ;
    FETCH NEXT FROM myCursor INTO @job_id, @name, @enabled;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
