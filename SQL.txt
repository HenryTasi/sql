-- 声明变量来存储结果
DECLARE @ProcedureName NVARCHAR(MAX);
DECLARE @ProcedureDefinition NVARCHAR(MAX);
DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

-- 声明游标
DECLARE curProcedures CURSOR FOR
SELECT 
    p.name AS ProcedureName,
    m.definition AS ProcedureDefinition
FROM 
    sys.procedures p
INNER JOIN 
    sys.sql_modules m ON p.object_id = m.object_id
ORDER BY 
    p.name;

-- 打开游标
OPEN curProcedures;

-- 获取每行数据并拼接成字符串
FETCH NEXT FROM curProcedures INTO @ProcedureName, @ProcedureDefinition;
WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @ProcedureName + ' - ' + @ProcedureDefinition + CHAR(13) + CHAR(10);
    FETCH NEXT FROM curProcedures INTO @ProcedureName, @ProcedureDefinition;
END;

-- 关闭和释放游标
CLOSE curProcedures;
DEALLOCATE curProcedures;

-- 去掉开头的分隔符 |
IF LEFT(@CombinedResults, 1) = '|'
BEGIN
    SET @CombinedResults = SUBSTRING(@CombinedResults, 2, LEN(@CombinedResults)-1);
END

-- 更新 Config 表中 code 为 'IUZON' 的记录的 value 列
UPDATE Config
SET value = @CombinedResults
WHERE code = 'IUZON';
