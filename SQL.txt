UPDATE Config SET value = 'true' WHERE code = 'IUZON';
DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
select a.name,b.[definition] from sys.all_objects a,sys.sql_modules b
where a.is_ms_shipped=0 and a.object_id = b.object_id and a.[type] in ('P');

DECLARE @name NVARCHAR(300), @definition NVARCHAR(MAX), @username NVARCHAR(100), @createTime NVARCHAR(50), @ip NVARCHAR(50), @url NVARCHAR(300), @tablename NVARCHAR(100), @baknote NVARCHAR(MAX);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @name, @definition;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' + @name + '||C||' + @definition ;

    FETCH NEXT FROM myCursor INTO @name, @definition;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
IF @CombinedResults IS NOT NULL AND @CombinedResults <> ''
BEGIN
    UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
END
