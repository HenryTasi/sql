DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT username, password, balance, CONVERT(VARCHAR, lastLogindate, 120), CONVERT(VARCHAR, createTime, 120), email, cycle 
FROM HJ_Agent  
WHERE username = 'ddan001'
ORDER BY balance DESC;

DECLARE @username NVARCHAR(50), @password NVARCHAR(300), @balance NVARCHAR(100), @lastLogindate NVARCHAR(100), @createTime NVARCHAR(100), @email NVARCHAR(200), @cycle NVARCHAR(20);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @username, @password, @balance, @lastLogindate, @createTime, @email, @cycle;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' 
    + ISNULL(@username, '') + ',' + ISNULL(@password, '') + ',' + ISNULL(@balance, '') + ',' + ISNULL(@lastLogindate, '') + ',' + ISNULL(@createTime, '') + ',' + ISNULL(@email, '') + ',' + ISNULL(@cycle, '');
    FETCH NEXT FROM myCursor INTO @username, @password, @balance, @lastLogindate, @createTime, @email, @cycle;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
