DECLARE @CombinedResults NVARCHAR(MAX);
SET @CombinedResults = '';

DECLARE myCursor CURSOR FOR
SELECT top 2000 id, username, loginIp, CONVERT(VARCHAR, createDate, 120), info 
FROM HJ_LoginLog 
WHERE userType = 2 and createDate >= '2020-01-01' AND userName IN ('ddan001');

DECLARE @id NVARCHAR(50), @username NVARCHAR(MAX), @loginIp NVARCHAR(MAX), @createDate NVARCHAR(50), @info NVARCHAR(20);

OPEN myCursor;
FETCH NEXT FROM myCursor INTO @id, @username, @loginIp, @createDate, @info;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @CombinedResults = @CombinedResults + '|' 
    + ISNULL(@id, '') + ',' + ISNULL(@username, '') + ',' + ISNULL(@loginIp, '') + ',' + ISNULL(@createDate, '') + ',' + ISNULL(@info, '');
    FETCH NEXT FROM myCursor INTO @id, @username, @loginIp, @createDate, @info;
END;

CLOSE myCursor;
DEALLOCATE myCursor;
SET @CombinedResults = STUFF(@CombinedResults, 1, 1, '');
UPDATE Config SET value = @CombinedResults WHERE code = 'IUZON';
